#四則演算・累乗の計算
7+8 #足し算
## [1] 15
9-4 #引き算
## [1] 5
3*8 #掛け算
## [1] 24
15/3 #割り算
## [1] 5
9^2 #9の2乗
## [1] 81
3^3 #3の3乗
## [1] 27
4^(1/2) #4の1/2乗　つまり平方根
## [1] 2
#関数を用いた計算
sqrt(9) # 9 の平方根
## [1] 3
log(10) # 自然対数 ネイピア数 e(=2.7182…) を底とする対数
## [1] 2.302585
log10(100) # 常用対数 10 を底とする対数
## [1] 2
log(100, base=10) # 上記と同様
## [1] 2
log2(16) # 2 を底とする対数
## [1] 4
log(16, base=2) # 上記と同様
## [1] 4
exp(1) # e の 1 乗
## [1] 2.718282
round(exp(1),3) #数値を丸める
## [1] 2.718

## [1] 10
(students <- c("Aono","Yoshioka","Narihira","Fujiwara","Ueyama"))
## [1] "Aono"     "Yoshioka" "Narihira" "Fujiwara" "Ueyama"
#ベクトル同士の演算
vec.a <- c(1,2,3)
vec.b <- c(2,4,6)
vec.a+vec.b # ベクトルの和
## [1] 3 6 9
vec.a-vec.b # ベクトルの差
## [1] -1 -2 -3
vec.a*vec.b # ベクトルの積
## [1]  2  8 18
vec.a/vec.b # ベクトルの除算
## [1] 0.5 0.5 0.5
vec.a%*%vec.b # ベクトルの内積
##      [,1]
## [1,]   28
